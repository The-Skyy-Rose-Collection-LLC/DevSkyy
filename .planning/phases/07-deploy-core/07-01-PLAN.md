```bash
# WP-CLI remote execution (line 83-90, proven working)
wp_remote() {
    local cmd="$1"
    if [[ "$DRY_RUN" == "true" ]]; then
        log_info "[DRY RUN] wp $cmd"
        return 0
    fi
    sshpass -p "$SSH_PASS" ssh -o StrictHostKeyChecking=accept-new "${SSH_USER}@${SSH_HOST}" "wp $cmd" 2>/dev/null
}

# Cache flush pattern (line 460-469, proven working)
do_flush_cache() {
    wp_remote "cache flush"
    wp_remote "transient delete --all"
    wp_remote "rewrite flush"
}
```
