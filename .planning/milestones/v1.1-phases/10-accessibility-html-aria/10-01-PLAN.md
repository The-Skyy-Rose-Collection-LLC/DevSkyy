import re, os, sys

def find_buttons_without_type(filepath):
    with open(filepath, 'r') as f:
        content = f.read()
    issues = []
    for m in re.finditer(r'<button\b', content):
        start = m.start()
        pos = m.end()
        in_php = False
        while pos < len(content):
            c = content[pos]
            if content[pos:pos+5] == '<?php':
                in_php = True
                pos += 5
                continue
            if in_php and content[pos:pos+2] == '?>':
                in_php = False
                pos += 2
                continue
            if in_php:
                pos += 1
                continue
            if c == '>':
                tag = content[start:pos+1]
                clean = re.sub(r'<\?php.*?\?>', '', tag, flags=re.DOTALL)
                if 'type=' not in clean:
                    line_num = content[:start].count('\n') + 1
                    issues.append(line_num)
                break
            pos += 1
    return issues

total = 0
for root, dirs, files in os.walk('.'):
    dirs[:] = [d for d in dirs if d not in ('node_modules', 'vendor')]
    for f in files:
        if f.endswith('.php'):
            path = os.path.join(root, f)
            issues = find_buttons_without_type(path)
            if issues:
                print(f'FAIL: {path} has buttons without type at lines {issues}')
                total += len(issues)

# Check handle collision
with open('inc/accessibility-seo.php', 'r') as f:
    a11y = f.read()
handles = re.findall(r\"wp_enqueue_(?:style|script)\(\s*['\"]([^'\"]+)['\"]\", a11y)
if len(handles) != len(set(handles)):

if total == 0:
