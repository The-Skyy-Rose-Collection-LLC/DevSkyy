```php
<?php
/**
 * Mega Menu Walker — exposes top-level items with optional flyout columns.
 *
 * Usage: pass via wp_nav_menu( array( 'walker' => new Skyyrose_Mega_Menu_Walker() ) ).
 * Top-level menu items with CSS class `mega-menu-trigger` render with data-mega="true"
 * + aria-haspopup="true"; the consuming JS (mobile-drawer.js / sticky-header.js) reads
 * those attrs and renders the flyout column container.
 *
 * @package SkyyRose
 */

defined( 'ABSPATH' ) || exit;

if ( ! class_exists( 'Walker_Nav_Menu' ) ) {
    require_once ABSPATH . WPINC . '/class-walker-nav-menu.php';
}

class Skyyrose_Mega_Menu_Walker extends Walker_Nav_Menu {

    public function start_el( &$output, $item, $depth = 0, $args = null, $id = 0 ): void {
        $is_mega = $depth === 0 && in_array( 'mega-menu-trigger', (array) ( $item->classes ?? array() ), true );
        $extra   = $is_mega ? ' data-mega="true" aria-haspopup="true" aria-expanded="false"' : '';

        $url   = ! empty( $item->url ) ? esc_url( $item->url ) : '#';
        $title = esc_html( $item->title ?? '' );
        $output .= '<li class="menu-item' . ( $is_mega ? ' menu-item--mega' : '' ) . '">';
        $output .= '<a href="' . $url . '"' . $extra . '>' . $title . '</a>';
    }

    public function end_el( &$output, $item, $depth = 0, $args = null ): void {
        $output .= '</li>';
    }
}
```

Create `wordpress-theme/skyyrose-flagship/assets/js/mobile-drawer.js`:
```javascript
/**
 * Mobile drawer — focus trap, outside-tap dismiss, Escape close.
 * Triggers on .drawer-trigger click; opens #skyyrose-mobile-drawer <dialog>.
 */
( function () {
    'use strict';

    var trigger = document.querySelector( '.drawer-trigger' );
    var dialog  = document.getElementById( 'skyyrose-mobile-drawer' );
    if ( ! trigger || ! dialog ) return;

    var closeBtn = dialog.querySelector( '.mobile-drawer__close' );
    var lastFocus = null;

    function getFocusables() {
        return dialog.querySelectorAll(
            'a[href], button:not([disabled]), [tabindex]:not([tabindex="-1"])'
        );
    }

    function open() {
        lastFocus = document.activeElement;
        // Native <dialog> showModal() handles backdrop + tab containment
        if ( typeof dialog.showModal === 'function' ) {
            dialog.showModal();
        } else {
            dialog.setAttribute( 'open', '' );
        }
        trigger.setAttribute( 'aria-expanded', 'true' );
        // Focus first focusable element inside drawer
        var focusables = getFocusables();
        if ( focusables.length ) focusables[ 0 ].focus();
    }

    function close() {
        if ( typeof dialog.close === 'function' ) {
            dialog.close();
        } else {
            dialog.removeAttribute( 'open' );
        }
        trigger.setAttribute( 'aria-expanded', 'false' );
        if ( lastFocus && typeof lastFocus.focus === 'function' ) {
            lastFocus.focus();
        }
    }

    trigger.addEventListener( 'click', open );
    closeBtn && closeBtn.addEventListener( 'click', close );

    // Outside-tap: <dialog> backdrop click target IS the dialog itself
    dialog.addEventListener( 'click', function ( e ) {
        var rect = dialog.getBoundingClientRect();
        var inside = e.clientX >= rect.left && e.clientX <= rect.right &&
                     e.clientY >= rect.top  && e.clientY <= rect.bottom;
        if ( ! inside ) close();
    } );

    // Escape close (showModal handles this natively, but legacy browsers don't)
    dialog.addEventListener( 'cancel', function ( e ) {
        e.preventDefault();
        close();
    } );

    // Close on internal link click (so navigation actually proceeds)
    dialog.querySelectorAll( '.mobile-drawer__nav a' ).forEach( function ( link ) {
        link.addEventListener( 'click', close );
    } );
}() );
```

```php
<?php
/**
 * Breadcrumbs JSON-LD emitter.
 *
 * Parallel to the microdata renderer at inc/seo.php:435. JSON-LD is what
 * Google Rich Results Test reads. The microdata version stays for ARIA +
 * non-Google crawlers.
 *
 * @package SkyyRose
 */

defined( 'ABSPATH' ) || exit;

/**
 * Build the JSON-LD string from a list of breadcrumb items.
 *
 * @param array<int, array{name: string, url: string}> $items Breadcrumb items in order.
 * @return string JSON-LD <script> body, or empty string if input is empty.
 */
function skyyrose_breadcrumbs_jsonld_build( array $items ): string {
    if ( empty( $items ) ) {
        return '';
    }

    $list = array();
    foreach ( $items as $i => $item ) {
        $list[] = array(
            '@type'    => 'ListItem',
            'position' => $i + 1,
            'name'     => wp_strip_all_tags( $item['name'] ?? '' ),
            'item'     => esc_url_raw( $item['url'] ?? '' ),
        );
    }

    $payload = array(
        '@context'        => 'https://schema.org',
        '@type'           => 'BreadcrumbList',
        'itemListElement' => $list,
    );

    return wp_json_encode( $payload, JSON_UNESCAPED_SLASHES );
}

/**
 * Build current-page breadcrumb items from WP context.
 *
 * @return array<int, array{name: string, url: string}>
 */
function skyyrose_breadcrumbs_jsonld_current_items(): array {
    $items = array(
        array( 'name' => __( 'Home', 'skyyrose' ), 'url' => home_url( '/' ) ),
    );

    if ( is_singular( 'product' ) ) {
        $product_id = get_queried_object_id();
        $terms      = get_the_terms( $product_id, 'product_cat' );
        if ( $terms && ! is_wp_error( $terms ) ) {
            $term = $terms[0];
            $items[] = array(
                'name' => $term->name,
                'url'  => get_term_link( $term ),
            );
        }
        $items[] = array( 'name' => get_the_title( $product_id ), 'url' => get_permalink( $product_id ) );
    } elseif ( is_tax( 'product_cat' ) || is_category() || is_tag() ) {
        $term    = get_queried_object();
        $items[] = array( 'name' => $term->name, 'url' => get_term_link( $term ) );
    } elseif ( is_page() ) {
        $page_id = get_queried_object_id();
        // Walk ancestors
        $ancestors = array_reverse( get_post_ancestors( $page_id ) );
        foreach ( $ancestors as $aid ) {
            $items[] = array( 'name' => get_the_title( $aid ), 'url' => get_permalink( $aid ) );
        }
        $items[] = array( 'name' => get_the_title( $page_id ), 'url' => get_permalink( $page_id ) );
    } elseif ( is_singular() ) {

    }

    return $items;
}

/**
 * Hook callback — emits the JSON-LD <script> tag in <head>.
 */
function skyyrose_breadcrumbs_jsonld_emit(): void {
    if ( is_front_page() || is_404() ) {
        return;
    }
    $items = skyyrose_breadcrumbs_jsonld_current_items();
    $json  = skyyrose_breadcrumbs_jsonld_build( $items );
    if ( '' === $json ) {
        return;
    }
    echo '<script type="application/ld+json">' . $json . '</script>' . "\n";
}
```

- [ ] **Step 1: Write sticky-header JS**

Create `wordpress-theme/skyyrose-flagship/assets/js/sticky-header.js`:
```javascript
/**
 * Scroll-aware sticky header.
 *
 * State machine:
 * - "at-top": page Y = 0 → transparent, no shadow
 * - "scrolled": page Y > threshold + scrolling DOWN → translateY(-100%) (hidden)
 * - "scrolled-up": page Y > threshold + scrolling UP → translateY(0) (visible, solid)
 */
( function () {
    'use strict';

    var header = document.querySelector( '[data-sticky-header]' );
    if ( ! header ) return;

    var lastY = window.scrollY;
    var threshold = 80; // px from top before hide-on-scroll-down activates
    var ticking = false;

    function update() {
        var y = window.scrollY;

        if ( y <= 0 ) {
            header.dataset.state = 'at-top';
        } else if ( y > lastY && y > threshold ) {
            header.dataset.state = 'scrolled-down';
        } else if ( y < lastY ) {
            header.dataset.state = 'scrolled-up';
        }

        lastY = y;
        ticking = false;
    }

    window.addEventListener( 'scroll', function () {
        if ( ! ticking ) {
            window.requestAnimationFrame( update );
            ticking = true;
        }
    }, { passive: true } );

    // Initialize state
    update();
}() );
```
